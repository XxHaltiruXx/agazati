# Modul√°ris sz√°mol√≥g√©p
def osszeadas(a, b):
    """K√©t sz√°m √∂sszege"""
    return a + b

def kivonas(a, b):
    """K√©t sz√°m k√ºl√∂nbs√©ge"""
    return a - b

def szorzas(a, b):
    """K√©t sz√°m szorzata"""
    return a * b

def osztas(a, b):
    """K√©t sz√°m h√°nyadosa"""
    if b == 0:
        return "Hiba: 0-val nem lehet osztani!"
    return a / b

def hatvanyozas(a, b):
    """Els≈ë sz√°m a m√°sodik hatv√°ny√°n"""
    return a ** b

# Sz√°mol√≥g√©p f√ºggv√©ny
def szamologep(muvelet, a, b):
    """
    Egyszer≈± sz√°mol√≥g√©p f√ºggv√©ny
    
    Param√©terek:
        muvelet (str): '+', '-', '*', '/', '**'
        a (float): Els≈ë sz√°m
        b (float): M√°sodik sz√°m
    """
    if muvelet == '+':
        return osszeadas(a, b)
    elif muvelet == '-':
        return kivonas(a, b)
    elif muvelet == '*':
        return szorzas(a, b)
    elif muvelet == '/':
        return osztas(a, b)
    elif muvelet == '**':
        return hatvanyozas(a, b)
    else:
        return "Ismeretlen m≈±velet!"

# Haszn√°lat
print(szamologep('+', 10, 5))   # 15
print(szamologep('-', 10, 5))   # 5
print(szamologep('*', 10, 5))   # 50
print(szamologep('/', 10, 5))   # 2.0
print(szamologep('**', 2, 8))   # 256

# Lista seg√©df√ºggv√©nyek
def lista_atlag(szamok):
    """Sz√°mok √°tlaga"""
    if len(szamok) == 0:
        return 0
    return sum(szamok) / len(szamok)

def lista_max_index(szamok):
    """Legnagyobb elem indexe"""
    if len(szamok) == 0:
        return -1
    max_ertek = max(szamok)
    return szamok.index(max_ertek)

def lista_parosak(szamok):
    """P√°ros sz√°mok sz≈±r√©se"""
    return [x for x in szamok if x % 2 == 0]

def lista_dupla(szamok):
    """Minden elem dupl√°z√°sa"""
    return [x * 2 for x in szamok]

# Haszn√°lat
szamok = [3, 7, 2, 9, 4, 6, 1]

print(f"√Åtlag: {lista_atlag(szamok)}")
print(f"Legnagyobb index: {lista_max_index(szamok)}")
print(f"P√°rosak: {lista_parosak(szamok)}")
print(f"Dupl√°zva: {lista_dupla(szamok)}")
