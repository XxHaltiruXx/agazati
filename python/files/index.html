M√≥d Le√≠r√°s Megjegyz√©s
"r" Olvas√°s (Read) Alap√©rtelmezett. Hiba, ha nem l√©tezik.
"w" √çr√°s (Write) L√©trehozza/fel√ºl√≠rja a f√°jlt.
"a" Hozz√°f≈±z√©s (Append) A v√©g√©re √≠r, l√©trehozza ha nincs.

"r+" Olvas√°s √©s √≠r√°s F√°jl elej√©t≈ël.
"b" Bin√°ris m√≥d Kombin√°ld: "rb", "wb"

from datetime import datetime

class Naplo:
    """Egyszer≈± napl√≥ kezel≈ë"""
    
    def __init__(self, fajlnev="naplo.txt"):
        self.fajlnev = fajlnev
    
    def ir(self, uzenet):
        """Bejegyz√©s √≠r√°sa id≈ëb√©lyeggel"""
        ido = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        bejegyzes = f"[{ido}] {uzenet}\n"
        
        with open(self.fajlnev, "a", encoding="utf-8") as fajl:
            fajl.write(bejegyzes)
        
        print(f"Napl√≥ba √≠rva: {uzenet}")
    
    def olvas(self):
        """Teljes napl√≥ olvas√°sa"""
        try:
            with open(self.fajlnev, "r", encoding="utf-8") as fajl:
                return fajl.read()
        except FileNotFoundError:
            return "A napl√≥ f√°jl m√©g nem l√©tezik."
    
    def utolso_n(self, n=10):
        """Utols√≥ N bejegyz√©s"""
        try:
            with open(self.fajlnev, "r", encoding="utf-8") as fajl:
                sorok = fajl.readlines()
                return "".join(sorok[-n:])
        except FileNotFoundError:
            return "A napl√≥ f√°jl m√©g nem l√©tezik."
    
    def torol(self):
        """Napl√≥ t√∂rl√©se"""
        import os
        if os.path.exists(self.fajlnev):
            os.remove(self.fajlnev)
            print("Napl√≥ t√∂r√∂lve")

# Haszn√°lat
naplo = Naplo("alkalmazas.log")

naplo.ir("Alkalmaz√°s elindult")
naplo.ir("Felhaszn√°l√≥ bejelentkezett")
naplo.ir("Adatok bet√∂ltve")
naplo.ir("M≈±velet v√©grehajtva")

print("\n=== Teljes napl√≥ ===")
print(naplo.olvas())

print("\n=== Utols√≥ 2 bejegyz√©s ===")
print(naplo.utolso_n(2))

import json
import os

class Konfiguracio:
    """Alkalmaz√°s konfigur√°ci√≥ kezel≈ë"""
    
    def __init__(self, fajlnev="config.json"):
        self.fajlnev = fajlnev
        self.config = self.betolt()
    
    def betolt(self):
        """Konfigur√°ci√≥ bet√∂lt√©se"""
        if os.path.exists(self.fajlnev):
            with open(self.fajlnev, "r", encoding="utf-8") as fajl:
                return json.load(fajl)
        else:
            # Alap√©rtelmezett konfigur√°ci√≥
            return {
                "tema": "sotet",
                "nyelv": "hu",
                "ertesitesek": True,
                "auto_mentes": True
            }
    
    def ment(self):
        """Konfigur√°ci√≥ ment√©se"""
        with open(self.fajlnev, "w", encoding="utf-8") as fajl:
            json.dump(self.config, fajl, ensure_ascii=False, indent=4)
        print("Konfigur√°ci√≥ mentve")
    
    def get(self, kulcs, alapertelmezett=None):
        """Be√°ll√≠t√°s lek√©rdez√©se"""
        return self.config.get(kulcs, alapertelmezett)
    
    def set(self, kulcs, ertek):
        """Be√°ll√≠t√°s m√≥dos√≠t√°sa"""
        self.config[kulcs] = ertek
        self.ment()
    
    def kiir(self):
        """Konfigur√°ci√≥ megjelen√≠t√©se"""
        print("\n=== Konfigur√°ci√≥ ===")
        for kulcs, ertek in self.config.items():
            print(f"{kulcs}: {ertek}")

# Haszn√°lat
config = Konfiguracio()

config.kiir()
print(f"\nT√©ma: {config.get('tema')}")

config.set("tema", "vilagos")
config.set("nyelv", "en")
config.kiir()

  • Ellen≈ërizd a f√°jl l√©tez√©s√©t olvas√°s el≈ëtt

  • A "w" m√≥d fel√ºl√≠rja a f√°jlt - haszn√°ld "a"-t hozz√°f≈±z√©shez
  • Haszn√°lj os.path.join()-t platform f√ºggetlen √∫tvonalakhoz
  • JSON jobb struktur√°lt adatokhoz, mint sima sz√∂veg
  • Bin√°ris f√°jlokn√°l mindig "b" m√≥dot haszn√°lj
